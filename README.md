## Install

```bash
npm install --save button-type
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'button-type'
import 'button-type/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
